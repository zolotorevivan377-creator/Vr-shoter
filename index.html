// Инициализация сцены const scene = new THREE.Scene(); const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000); const renderer = new THREE.WebGLRenderer({ antialias: true }); renderer.setSize(window.innerWidth, window.innerHeight); document.body.appendChild(renderer.domElement); // Инициализация VR контекста let xrSession; let xrReferenceSpace; const initVR = async () => { try { xrSession = await navigator.xr.requestSession('immersive-vr', { requiredFeatures: ['local-floor'], }); xrReferenceSpace = await xrSession.requestReferenceSpace('local-floor'); renderer.xr.enabled = true; renderer.xr.setSession(xrSession); } catch (error) { console.error('Error initializing VR:', error); } }; initVR(); // Генерация уровня const generateLevel = () => { // Генерация случайных врагов const enemies = []; for (let i = 0; i < 10; i++) { const enemy = new THREE.Mesh( new THREE.BoxGeometry(1, 2, 1), new THREE.MeshBasicMaterial({ color: 0xff0000 }) ); enemy.position.set( Math.random() * 20 - 10, Math.random() * 2 + 1, Math.random() * 20 - 10 ); enemies.push(enemy); scene.add(enemy); } return enemies; }; // Движение игрока let playerVelocity = new THREE.Vector3(); const playerSpeed = 5; const handlePlayerInput = () => { const inputState = { forward: false, back: false, left: false, right: false, }; if (xrSession) { const inputSources = xrSession.inputSources; for (const inputSource of inputSources) { if (inputSource.gamepad) { const axes = inputSource.gamepad.axes; inputState.forward = axes[1] < -0.5; inputState.back = axes[1] > 0.5; inputState.left = axes[0] < -0.5; inputState.right = axes[0] > 0.5; } } } playerVelocity.set(0, 0, 0); if (inputState.forward) { playerVelocity.z -= playerSpeed; } if (inputState.back) { playerVelocity.z += playerSpeed; } if (inputState.left) { playerVelocity.x -= playerSpeed; } if (inputState.right) { playerVelocity.x += playerSpeed; } }; // Обновление игры const clock = new THREE.Clock(); let currentLevel = 0; let enemies = generateLevel(); const animate = () => { renderer.setAnimationLoop(animate); const delta = clock.getDelta(); handlePlayerInput(); camera.position.add(playerVelocity.multiplyScalar(delta)); // Обновление врагов enemies.forEach((enemy) => { enemy.position.x += Math.sin(clock.elapsedTime * 2) * 0.1; enemy.position.z += Math.cos(clock.elapsedTime * 2) * 0.1; }); // Проверка, все ли враги убиты if (enemies.length === 0) { currentLevel++; console.log(`Level ${currentLevel} complete!`); enemies = generateLevel(); } renderer.render(scene, camera); }; animate();
